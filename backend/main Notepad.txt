# SHATAI Backend 
from fastapi import FastAPI

app = FastAPI(
    title="SHATAI Health Platform",
    description="Sun'iy intellekt asosidagi sog'liqni baholash platformasi",
    version="1.0.0"
)

@app.get("/")
def root():
    return {
        "message": "SHATAI API ishga tushdi!",
        "status": "success",
        "version": "1.0.0"
    }

@app.get("/hello/{name}")
def say_hello(name: str):
    return {
        "message": f"Salom {name}!",
        "welcome": "SHATAI platformasiga xush kelibsiz"
    }
